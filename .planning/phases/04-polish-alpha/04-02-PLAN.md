**scripts/build-release.sh:**
```bash
#!/bin/bash
VERSION=${1:-dev}
mkdir -p dist

# Build for multiple platforms
GOOS=linux GOARCH=amd64 go build -o dist/percepta-linux-amd64 ./cmd/percepta
GOOS=linux GOARCH=arm64 go build -o dist/percepta-linux-arm64 ./cmd/percepta
GOOS=darwin GOARCH=amd64 go build -o dist/percepta-darwin-amd64 ./cmd/percepta
GOOS=darwin GOARCH=arm64 go build -o dist/percepta-darwin-arm64 ./cmd/percepta
GOOS=windows GOARCH=amd64 go build -o dist/percepta-windows-amd64.exe ./cmd/percepta

# Create tarballs
tar -czf dist/percepta-${VERSION}-linux-amd64.tar.gz -C dist percepta-linux-amd64
tar -czf dist/percepta-${VERSION}-linux-arm64.tar.gz -C dist percepta-linux-arm64
tar -czf dist/percepta-${VERSION}-darwin-amd64.tar.gz -C dist percepta-darwin-amd64
tar -czf dist/percepta-${VERSION}-darwin-arm64.tar.gz -C dist percepta-darwin-arm64
zip dist/percepta-${VERSION}-windows-amd64.zip dist/percepta-windows-amd64.exe
```
